"""
LeHome 业务系统 API 客户端
对接 LeHome Spring Boot 后端，所有业务数据通过 REST API 获取/写入
"""
import requests
import logging

logger = logging.getLogger("lehome_client")


class LeHomeClient:
    """LeHome 业务 API 调用客户端"""

    def __init__(self, base_url: str, token: str = ""):
        """
        :param base_url: LeHome API 地址，如 "http://192.168.100.96:8092"
        :param token: API 认证 Token（如果有）
        """
        self.base_url = base_url.rstrip("/")
        self.token = token
        self.timeout = 10  # 秒

    def _headers(self) -> dict:
        h = {"Content-Type": "application/json"}
        if self.token:
            h["Authorization"] = f"Bearer {self.token}"
        return h

    def _get(self, path: str, params: dict = None) -> dict:
        """通用 GET 请求"""
        try:
            resp = requests.get(
                f"{self.base_url}{path}",
                params=params,
                headers=self._headers(),
                timeout=self.timeout,
            )
            resp.raise_for_status()
            return resp.json()
        except requests.Timeout:
            logger.error(f"LeHome API 超时: GET {path}")
            return {"error": "请求超时，请稍后再试"}
        except requests.RequestException as e:
            logger.error(f"LeHome API 错误: GET {path} -> {e}")
            return {"error": str(e)}

    def _post(self, path: str, data: dict = None) -> dict:
        """通用 POST 请求"""
        try:
            resp = requests.post(
                f"{self.base_url}{path}",
                json=data,
                headers=self._headers(),
                timeout=self.timeout,
            )
            resp.raise_for_status()
            return resp.json()
        except requests.Timeout:
            logger.error(f"LeHome API 超时: POST {path}")
            return {"error": "请求超时，请稍后再试"}
        except requests.RequestException as e:
            logger.error(f"LeHome API 错误: POST {path} -> {e}")
            return {"error": str(e)}

    # =========================================================
    # 项目相关
    # =========================================================

    def search_project(self, keyword: str) -> dict:
        """
        模糊搜索项目
        GET /api/wx/project/search?q={keyword}
        返回示例：
        {
            "code": 0,
            "data": [
                {"id": 123, "name": "南京金茂府", "contactName": "张经理",
                 "contactInfo": "13800001111", "address": "南京市建邺区xxx",
                 "level": 2, "levelText": "银级"}
            ]
        }
        """
        return self._get("/api/wx/project/search", {"q": keyword})

    def get_project(self, project_id: int) -> dict:
        """
        获取项目详情
        GET /api/wx/project/{id}
        """
        return self._get(f"/api/wx/project/{project_id}")

    # =========================================================
    # 工单相关
    # =========================================================

    def get_repairs(self, project_id: int, limit: int = 5) -> dict:
        """
        查询项目最近工单
        GET /api/wx/repair/list?projectId={pid}&limit={n}
        返回示例：
        {
            "code": 0,
            "data": [
                {"id": 10001, "type": "叶黄", "desc": "前台绿植叶子发黄",
                 "createDate": "2026-04-10", "opState": 1, "opStateText": "已处理"}
            ]
        }
        """
        return self._get("/api/wx/repair/list", {
            "projectId": project_id, "limit": limit
        })

    def create_repair(self, project_id: int, repair_type: str,
                      desc: str, location: str = "", pic_url: str = "") -> dict:

        创建投诉工单
        POST /api/wx/repair/create
        Body: {"projectId": 123, "type": "叶黄", "desc": "...", "location": "前台"}
        返回示例：
        {"code": 0, "data": {"id": 10002, "repairNo": "R20260414001"}}

            "projectId": project_id,

            "desc": desc,

            "projectId": project_id, "limit": limit

"""
微信客户绑定服务
管理微信聊天窗口名 ↔ LeHome 项目ID 的映射关系
数据存储：本地 SQLite（wx_binding.db）
"""
import sqlite3
import os
import logging
from datetime import datetime

logger = logging.getLogger("binding_service")


class BindingService:
    """微信客户 ↔ 项目绑定管理"""

    def __init__(self, db_path: str = None):
        """
        :param db_path: SQLite 数据库路径，默认为当前目录 wx_binding.db
        """
        if db_path is None:
            _base = os.path.dirname(os.path.abspath(__file__))
            db_path = os.path.join(_base, "wx_binding.db")
        self.db_path = db_path
        self._init_db()

    def _init_db(self):
        """创建表（如果不存在）"""
        conn = sqlite3.connect(self.db_path)
        try:
            conn.execute("""
                CREATE TABLE IF NOT EXISTS wx_binding (
                    chat_name TEXT PRIMARY KEY,
                    project_id INTEGER NOT NULL,
                    project_name TEXT,
                    contact_name TEXT,
                    contact_info TEXT,
                    bind_time TEXT DEFAULT (datetime('now', 'localtime')),
                    is_active INTEGER DEFAULT 1
                )
            """)
            conn.execute("""
                CREATE TABLE IF NOT EXISTS wx_binding_history (
                    id INTEGER PRIMARY KEY AUTOINCREMENT,
                    chat_name TEXT NOT NULL,
                    project_id INTEGER NOT NULL,
                    project_name TEXT,
                    action TEXT,
                    action_time TEXT DEFAULT (datetime('now', 'localtime'))
                )
            """)
            conn.commit()
        finally:
            conn.close()

    def _conn(self) -> sqlite3.Connection:
        conn = sqlite3.connect(self.db_path)
        conn.row_factory = sqlite3.Row
        return conn

    def bind_customer(self, chat_name: str, project_id: int,
                      project_name: str = "", contact_name: str = "",
                      contact_info: str = "") -> dict:
        """绑定客户到项目"""
        conn = self._conn()
        try:
            conn.execute("""
                INSERT OR REPLACE INTO wx_binding
                (chat_name, project_id, project_name, contact_name, contact_info, is_active)
                VALUES (?, ?, ?, ?, ?, 1)
            """, (chat_name, project_id, project_name, contact_name, contact_info))
            conn.execute("""
                INSERT INTO wx_binding_history (chat_name, project_id, project_name, action)
                VALUES (?, ?, ?, 'bind')
            """, (chat_name, project_id, project_name))
            conn.commit()
            logger.info(f"绑定成功: {chat_name} → {project_name}(ID:{project_id})")
            return {"success": True, "message": "绑定成功"}
        finally:
            conn.close()

    def unbind_customer(self, chat_name: str) -> dict:
        """解绑客户"""
        conn = self._conn()
        try:
            row = conn.execute(
                "SELECT * FROM wx_binding WHERE chat_name = ? AND is_active = 1",
                (chat_name,)
            ).fetchone()
            if not row:
                return {"success": False, "message": "未找到绑定记录"}
            conn.execute(
                "UPDATE wx_binding SET is_active = 0 WHERE chat_name = ?",
                (chat_name,)
            )
            conn.execute("""
                INSERT INTO wx_binding_history (chat_name, project_id, project_name, action)
                VALUES (?, ?, ?, 'unbind')
            """, (row["chat_name"], row["project_id"], row["project_name"]))
            conn.commit()
            return {"success": True, "message": "解绑成功"}
        finally:
            conn.close()

    def get_binding(self, chat_name: str) -> dict | None:
        """查询绑定信息，返回 dict 或 None"""
        conn = self._conn()
        try:
            row = conn.execute(
                "SELECT * FROM wx_binding WHERE chat_name = ? AND is_active = 1",
                (chat_name,)
            ).fetchone()
            return dict(row) if row else None
        finally:
            conn.close()

    def is_bound(self, chat_name: str) -> bool:
        """判断是否已绑定"""
        return self.get_binding(chat_name) is not None
```

class AgentToolExecutor:
    """
    执行 GPT 选择的工具，返回结果供 GPT 生成最终回复。
    
    每个 execute 调用对应一次 tool_call，返回的 dict/list 会被
    json.dumps 后作为 tool message 送回 GPT。
    """

    def __init__(self, lehome_client, binding_service, wx_bot=None):
        """
        :param lehome_client: LeHomeClient 实例
        :param binding_service: BindingService 实例
        :param wx_bot: WXBot 实例（用于转人工建群等微信操作）
        """
        self.lehome = lehome_client
        self.binding = binding_service
        self.wx_bot = wx_bot

    def execute(self, tool_name: str, arguments: dict, chat_name: str) -> dict:
        """
        执行指定工具，返回结果 dict

        :param tool_name: GPT 选择的工具名
        :param arguments: GPT 传入的参数 dict
        :param chat_name: 当前微信聊天窗口名（chat.who）
        :return: 执行结果 dict，会被序列化为 JSON 送回 GPT
        """
        try:
            if tool_name == "search_project":
                return self._search_project(arguments)

            elif tool_name == "bind_project":
                return self._bind_project(arguments, chat_name)

            elif tool_name == "create_repair":
                return self._create_repair(arguments, chat_name)

            elif tool_name == "query_repairs":
                return self._query_repairs(chat_name)

            elif tool_name == "query_patrols":
                return self._query_patrols(chat_name)

            elif tool_name == "transfer_human":
                return self._transfer_human(arguments, chat_name)

            else:
                return {"error": f"未知工具: {tool_name}"}

        except Exception as e:
            logger.error(f"工具执行失败 [{tool_name}]: {e}")
            return {"error": f"操作失败: {str(e)}"}

# =====================================================================
# Agent 循环核心函数
# =====================================================================
def run_agent_loop(
    message: str,
    system_prompt: str,
    history: list,
    api_key: str,
    base_url: str,
    model: str,
    tool_executor: "AgentToolExecutor",
    chat_name: str,
    max_rounds: int = 5,
) -> str:
    """
    执行 Agent 循环：调用 GPT → 如果返回 tool_calls → 执行工具 → 送回结果 → 重复
    直到 GPT 返回纯文本回复或达到最大轮次。

    :param message: 用户当前消息
    :param system_prompt: 系统提示词（含客户状态注入）
    :param history: 对话历史 list[dict]，格式 [{"role": "user/assistant", "content": "..."}]
    :param api_key: 紫喵AI API Key
    :param base_url: API 端点基础 URL（https://newapi.aisonnet.org）
    :param model: 模型名称（gpt-5.2-chat-latest）
    :param tool_executor: AgentToolExecutor 实例
    :param chat_name: 微信聊天窗口名
    :param max_rounds: 最大工具调用轮次（防止无限循环）
    :return: 最终回复文本
    """
    import requests as _requests

    # 构建 messages
    messages = []
    if system_prompt:
        messages.append({"role": "system", "content": system_prompt})
    
    # 历史消息（从 SiverWXBot 的 memory_manager 格式转换）
    for h in history:
        role = "assistant" if h.get("attr") == "self" else "user"
        t = h.get("time", "")
        raw = h.get("content", "")
        sender = h.get("sender", "")
        if role == "user" and sender:
            content = f"[{t}] {sender}: {raw}" if t else f"{sender}: {raw}"
        else:
            content = f"[{t}] {raw}" if t else raw
        messages.append({"role": role, "content": content})
    
    # 当前用户消息
    messages.append({"role": "user", "content": message})

    # Agent 循环
    headers = {
        "Authorization": f"Bearer {api_key}",
        "Content-Type": "application/json",
    }
    api_endpoint = f"{base_url.rstrip('/')}/v1/chat/completions"

    for round_num in range(max_rounds):
        payload = {
            "model": model,
            "messages": messages,
            "tools": AGENT_TOOLS,
            "tool_choice": "auto",
        }

        try:
            resp = _requests.post(
                api_endpoint, headers=headers, json=payload, timeout=120
            )
            resp.raise_for_status()
            data = resp.json()
        except Exception as e:
            logger.error(f"Agent 循环第 {round_num+1} 轮 API 调用失败: {e}")
            return "抱歉，系统暂时繁忙，请稍后再试~"

        choice = data.get("choices", [{}])[0]
        finish_reason = choice.get("finish_reason", "")
        assistant_msg = choice.get("message", {})

        # 将 assistant 消息加入 messages（含 tool_calls 信息）
        messages.append(assistant_msg)

        if finish_reason == "tool_calls" or assistant_msg.get("tool_calls"):
            # GPT 决定调用工具
            tool_calls = assistant_msg.get("tool_calls", [])
            for tc in tool_calls:
                fn_name = tc["function"]["name"]
                fn_args_str = tc["function"]["arguments"]
                tc_id = tc["id"]

                try:
                    fn_args = json.loads(fn_args_str)
                except json.JSONDecodeError:
                    fn_args = {}

                logger.info(f"Agent 工具调用: {fn_name}({fn_args})")

                # 执行工具
                tool_result = tool_executor.execute(fn_name, fn_args, chat_name)

                # 将工具结果作为 tool message 送回
                messages.append({
                    "role": "tool",
                    "tool_call_id": tc_id,
                    "content": json.dumps(tool_result, ensure_ascii=False),
                })

            # 继续循环，让 GPT 处理工具结果
            continue

        else:
            # GPT 返回纯文本回复，Agent 循环结束
            reply = assistant_msg.get("content", "")
            if reply:
                return reply
            else:
                return "抱歉，我暂时无法回复，请稍后再试~"

    # 超过最大轮次
    logger.warning(f"Agent 循环超过 {max_rounds} 轮，强制返回")
    last_content = messages[-1].get("content", "")
    return last_content if last_content else "操作有点复杂，让我整理一下，请稍等~"
```

## 六、错误处理与边界情况

| 场景 | 处理方式 |
|------|---------|
| LeHome API 超时 | lehome_client._get/_post 捕获 Timeout，返回 `{"error": "请求超时"}` |
| LeHome API 500 | 捕获 HTTPError，返回错误信息，GPT 会据此告知客户稍后再试 |
| GPT API 调用失败 | run_agent_loop 捕获异常，返回兜底文案 |
| Agent 循环超过 5 轮 | 强制返回最后一条 content 或兜底文案 |
| 客户未绑定就要建工单 | create_repair 执行器检测无绑定，返回错误提示，GPT 据此引导绑定 |
| search_project 无结果 | API 返回空列表，GPT 会提示客户提供更多信息 |
| SQLite 数据库损坏 | binding_service 初始化时自动建表，异常时记录日志 |
